use Illuminate\Support\Facades\Http;

public function report(Throwable $exception)
{
    Http::withHeaders([
        'Content-Type' => 'application/json',
        'X-API-KEY' => env('NOTIFY_X_API_KEY'),
    ])->post(env('NOTIFY_URL'), [
        'message' => $exception->getMessage(),
        'details' => $exception->getTraceAsString(),
    ]);

    parent::report($exception);
}
package main

import (
    "bytes"
    "encoding/json"
    "log"
    "net/http"
    "os"
)

func notifyError(err error) {
    payload := map[string]string{
        "message": err.Error(),
        "details": "", // You might want to add stack trace here
    }

    jsonPayload, _ := json.Marshal(payload)

    req, _ := http.NewRequest("POST", os.Getenv("NOTIFY_URL"), bytes.NewBuffer(jsonPayload))
    req.Header.Set("Content-Type", "application/json")
    req.Header.Set("X-API-KEY", os.Getenv("NOTIFY_X_API_KEY"))

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        log.Println("Error sending notification:", err)
    }
    defer resp.Body.Close()
}

func main() {
    // Use notifyError in your error handling
    if err := someFunction(); err != nil {
        notifyError(err)
    }
}

import json
import traceback
from django.conf import settings
import requests

class NotifyMiddleware:
    def __init__(self, get_response):
        self.get_response = get_response

    def __call__(self, request):
        response = self.get_response(request)
        return response

    def process_exception(self, request, exception):
        payload = {
            'message': str(exception),
            'details': traceback.format_exc()
        }

        headers = {
            'Content-Type': 'application/json',
            'X-API-KEY': settings.NOTIFY_X_API_KEY
        }

        requests.post(settings.NOTIFY_URL, json=payload, headers=headers)

# In config/initializers/notify.rb
require 'net/http'
require 'json'

module Notify
  class << self
    def notify_error(exception)
      payload = {
        message: exception.message,
        details: exception.backtrace.join("\n")
      }

      uri = URI(ENV['NOTIFY_URL'])
      http = Net::HTTP.new(uri.host, uri.port)
      http.use_ssl = (uri.scheme == 'https')

      request = Net::HTTP::Post.new(uri.path, 'Content-Type' => 'application/json')
      request['X-API-KEY'] = ENV['NOTIFY_X_API_KEY']
      request.body = payload.to_json

      response = http.request(request)
      Rails.logger.info("Notify response: #{response.code} #{response.message}")
    end
  end
end
