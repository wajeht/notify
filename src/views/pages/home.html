⚡️ Notify in a Flash!

Create an app, add some channels, and boom, you're done! Notify can be embedded into any software that can make an HTTP request!

const express = require('express');
const app = express();

// at the very last
app.use(async (err, req, res, next) => {
  await fetch(process.env.NOTIFY_URL, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-API-KEY': process.env.NOTIFY_X_API_KEY,
    },
    body: JSON.stringify({
      appId: process.env.NOTIFY_APP_ID,
      message: err.message,
      details: err.stack,
    }),
  });

  res.status(500).json({ error: 'An error occurred' });
});

app.listen(3000, () => console.log('Server running on port 3000'));
// In app/Exceptions/Handler.php
use Illuminate\Support\Facades\Http;

public function report(Throwable $exception)
{
    Http::withHeaders([
        'Content-Type' => 'application/json',
        'X-API-KEY' => env('NOTIFY_X_API_KEY'),
    ])->post(env('NOTIFY_URL'), [
        'appId' => env('NOTIFY_APP_ID'),
        'message' => $exception->getMessage(),
        'details' => $exception->getTraceAsString(),
    ]);

    parent::report($exception);
}
